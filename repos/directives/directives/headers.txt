char* concat(char* a, char* b)
{
	int r1 = 0, r2 = 0;
	while (a[r1] != '\0')
	{
		r1++;
	}
	while (b[r2] != '\0')
	{
		r2++;
	}
	char* c = calloc(r1 + r2, sizeof(char));
	for (int i = 0; i < r1; i++) {
		c[i] = a[i];
	}
	for (int i = 0; i < r2; i++) {
		c[r1 + i] = b[i];
	}
	c[r2 + r1] = '\0';
	return c;
}
